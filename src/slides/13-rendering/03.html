Font Smoothing

this is a text paragraph

 
            
              .example {
                // Turns font smoothing off
                // Displays text with jagged sharp edges
                -webkit-font-smoothing: none;
                -moz-osx-font-smoothing: none;
              }

this is a text paragraph

 
            
              .example {
                // Smooth the font on the level of the pixel
                // For light text on dark backgrounds makes it look lighter
                -webkit-font-smoothing: antialiased;
              }

this is a text paragraph

 
            
              .example {
                // On non-retina displays text will be sharper
                // (the risk is to have fonts looking a bit bolder)
                -webkit-font-smoothing: subpixel-antialiased;
                -moz-osx-font-smoothing: greyscale;
              }
