Letter Spacing

this is a text paragraph

 
            
              .example {
                // Font Kerning information in the font
                // will be applied
                letter-spacing: normal;
              }

this is a text paragraph

 
            
              .example {
                // Defines an extra space between characters
                letter-spacing: 0.5rem;
              }

this is a text paragraph

 
            
              .example {
                // Negative values are allowed
                letter-spacing: -2px;
              }

this is a text paragraph

 
            
              .example {
                // Portions of pixels are allowed
                letter-spacing: 0.75px;
              }
