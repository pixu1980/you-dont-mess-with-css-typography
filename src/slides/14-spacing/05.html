Word Spacing

this is a text paragraph

 
            
              .example {
                // Defines normal space between words
                // (default: 0.25em)
                word-spacing: normal;
              }

this is a text paragraph

 
            
              .example {
                // Defines an extra space between characters
                word-spacing: 1rem;
              }

this is a text paragraph

 
            
              .example {
                // Negative values are allowed
                word-spacing: -4px;
              }

this is a text paragraph

 
            
              .example {
                // Portions of pixels are allowed
                word-spacing: 1.75px;
              }
