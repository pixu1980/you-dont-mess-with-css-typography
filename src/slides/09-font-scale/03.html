            @use 'sass:math';
            
            $font-ratio: 1.25; /* major third 4:5 */
            /* 
              16px/10px = 0.625, expressed in em
              makes the font-size reactive to browser settings
              and makes 1rem = 10px,
              allowing to use rems as a simple division base-10 value
            */
            $font-root: 0.625em;      
            $font-base: 1.6rem;
  
            html {
              font-size: $font-root;
              color: white;
            }
            
            body {
              font-size: $font-base;
              line-height: $font-ratio;
            }
            
            h1, h2, h3, h4, h5, h6, p {
              margin: calc(#{$font-base} * #{$font-ratio}) 0 0;
            }
